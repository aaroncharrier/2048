def start():
    game = np.zeros((4, 4), np.int64)
    temp = np.where(game == 0)
    zeros = []
    for i, row in enumerate(temp[0]):
        zeros.append([row, temp[1][i]])
    choice = random.choice(zeros)
    number = random.choice([2, 4])
    game[choice[0]][choice[1]] = number
    temp = np.where(game == 0)
    zeros = []
    for i, row in enumerate(temp[0]):
        zeros.append([row, temp[1][i]])
    choice = random.choice(zeros)
    number = random.choice([2, 4])
    game[choice[0]][choice[1]] = number
    return game

        choice = random.choice(zeros)
        game[choice[0], choice[1]] = random.choice([2, 4])
        return game, False

def left(game):
    new_reward = 0
    shape = game.shape
    temp = np.where(game == 0)
    counts = Counter(temp[0])
    for row in range(shape[0]):
        temp_list = game[row].tolist()
        if counts[row] != shape[0]:
            temp = [i for i in temp_list if i != 0 ]
            while len(temp) < shape[0]:
                temp.append(0)
            temp_list = temp
        for i, col in enumerate(temp_list[:-1]):
            col = temp_list[i]
            if col == temp_list[i + 1]:
                new_reward += col * 2
                temp_list[i] = col * 2
                temp_list[i + 1] = 0
        if counts[row] != shape[0]:
            temp = [i for i in temp_list if i != 0 ]
            while len(temp) < shape[0]:
                temp.append(0)
            temp_list = temp
        game[row] = temp_list
    return game, new_reward

def find_best_dir(board_try, attempts):
    direction = {'DOWN': 0, 'UP': 0, 'LEFT': 0, 'RIGHT': 0}
    for dir in ['DOWN', 'UP', 'LEFT', 'RIGHT']:
        for i in range(attempts):
            # if i % 100 == 0:
            #     print(i, direction)
            available_choices = [dir]
            board = board_try.copy()
            turn = 0
            failed = False
            while not failed:
                reward = 0
                start_board = board.copy()
                temp_choice = random.choice(available_choices)
                available_choices.remove(temp_choice)
                if temp_choice == 'DOWN':
                    board, reward = down(board)
                elif temp_choice == 'UP':
                    board, reward = up(board)
                elif temp_choice == 'LEFT':
                    board, reward = left(board)
                elif temp_choice == 'RIGHT':
                    board, reward = right(board)
                temp123 = []
                for i in [0, 1, 2, 3]:
                    for ii in [0, 1, 2, 3]:
                        temp123.append(start_board[i][ii] == board[i][ii])
                if not np.all(temp123):
                    turn += 1
                    board, failed = new_tiles(board, turn)
                    available_choices = ['DOWN', 'UP', 'LEFT', 'RIGHT']
                if available_choices == []:
                    failed = True
            direction[dir] += turn
#     print(sorted(direction.items(), key=lambda item: item[1], reverse=True))
    return list(dict(sorted(direction.items(), key=lambda item: item[1], reverse=True)).keys())[0], direction

The goal we are trying to achieve is the largest 'max' number that we can in a playthrough. 'max' is positively correlated with try_attempts, score, turns, and time.

Focusing on time: Does a larger 'time' value cause a larger 'max' number? No, time is derived from the number of 'try_attempts' that the script runs through to determine the direction to go. Increasing the 'try_attempts' will increase the time is takes to play a game.

'score' and 'turns' are closely correlated. Our current script focusses on 'turns' to decide which direction to go. Can we shift the focus to 'score' will the result be different? For example the best one move that generates the highest score or a playthrough of 'x' games to determine the highest scoring direction to go.
